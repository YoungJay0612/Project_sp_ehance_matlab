168function y = lnshift(x,t)
169% lnshift -- t circular left shift of 1-d signal
170%  Usage
171%    y = lnshift(x,t)
172%  Inputs
173%    x   1-d signal
174%  Outputs
175%    y   1-d signal
176%        y(i) = x(i+t) for i+t < n
177%           y(i) = x(i+t-n) else
178%
179% Copyright (c) 2000. Prof Israel Cohen.
180% All rights reserved. 
181% ***************************************************************@
182szX=size(x);
183if szX(1)>1
184    n=szX(1);
185    y=[x((1+t):n); x(1:t)];
186else
187    n=szX(2);
188    y=[x((1+t):n) x(1:t)];
189end
